```swift
class Animal {
    let brain = true
    var legs = 0
}

class Human: Animal {
    override var legs: Int {
        get{
            return super.legs
        }
        set{
            super.legs = 2
        }
    }
}

class Pet: Animal {
    var fleas = 0
    override var legs: Int {
        get{
            return super.legs
        }
        set{
            super.legs = 4
        }
    }
}

class Dog: Pet {
    override var fleas: Int {
        get{
            return super.fleas
        }
        set{
            super.fleas = 8
        }
    }
}

class Cat: Pet {
    override var fleas: Int {
        get{
            return super.fleas
        }
        set{
            super.fleas = 4
        }
    }
}
```
